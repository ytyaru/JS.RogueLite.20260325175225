### 1. ファイルの提示 (1/1)

#### `src/js/lib/utils/dom/scene-manager.js`

```javascript
import { Typer } from '../typer.ja.js';

/**
 * 場面管理システム専用の例外
 */
class SceneError extends Error {
    constructor(message, options) {
        super(message, options);
        this.name = "SceneError";
    }
}

/**
 * 場面を構成する個別の部品
 */
class SceneItem {
    #id; #el; #build; #render;
    constructor(conf) {
        Typer.is(Object, conf, 'SceneItem config');
        Typer.is(String, conf.id, 'id');
        Typer.is(HTMLElement, conf.el, 'el');
        if (conf.build) Typer.is(Function, conf.build, 'build');
        if (conf.render) Typer.is(Function, conf.render, 'render');

        this.#id = conf.id;
        this.#el = conf.el;
        this.#build = conf.build;
        this.#render = conf.render;

        this.#el.setAttribute('data-scene-parts-id', this.#id);
        if (this.#build) this.#build(this.#el);
    }
    get id() { return this.#id; }
    get el() { return this.#el; }
    render(...args) {
        if (this.#render) this.#render(this.#el, ...args);
    }
}

/**
 * 場面（画面全体）の定義
 */
class Scene {
    #id; #el; #build; #render; #setup; #items = new Map();
    constructor(conf) {
        Typer.is(Object, conf, 'Scene config');
        Typer.is(String, conf.id, 'id');
        this.#id = conf.id;
        this.#el = conf.el || document.createElement('div');
        Typer.is(HTMLElement, this.#el, 'el');

        this.#build = conf.build;
        this.#render = conf.render;
        this.#setup = conf.setup;

        if (conf.items) {
            Typer.is(Array, conf.items, 'items');
            this.#registerItems(conf.items);
        }

        this.#el.setAttribute('data-scene-id', this.#id);
        this.#el.classList.add('is-hidden');
        this.#el.style.contentVisibility = 'hidden';
    }

    get id() { return this.#id; }
    get el() { return this.#el; }
    get items() { return this.#items; }

    #registerItems(items) {
        items.forEach(i => {
            if (this.#items.has(i.id)) {
                throw new SceneError(`場面 "${this.#id}" 内で部品ID "${i.id}" が重複しています。一意なIDを指定してください。`);
            }
            this.#items.set(i.id, new SceneItem(i));
        });
    }

    build() { if (this.#build) this.#build(this); }
    setup() { if (this.#setup) this.#setup(this); }
    
    render(data = {}) {
        if (data.items) {
            data.items.forEach(({id, args}) => {
                const item = this.#items.get(id);
                if (!item) throw new SceneError(`場面 "${this.#id}" に部品ID "${id}" は存在しません。`);
                item.render(...args);
            });
        }
        if (this.#render) this.#render(this, data.args || []);
    }
}

/**
 * 場面全体の管理者
 */
export class SceneManager {
    static #configs = new Map();
    static #instances = new Map();
    static #isBuilt = false;

    static add(config) {
        if (this.#isBuilt) throw new SceneError("構築(build)後は場面を追加できません。新しく登録するには clear() でリセットしてください。");
        Typer.is(Object, config, 'config');
        Typer.is(String, config.id, 'id');
        
        if (this.#configs.has(config.id)) {
            throw new SceneError(`場面ID "${config.id}" は既に登録されています。一意なIDを指定してください。`);
        }
        this.#configs.set(config.id, config);
    }

    static build() {
        if (this.#isBuilt) throw new SceneError("既に構築済みです。再度実行するには clear() でリセットしてください。");
        if (this.#configs.size === 0) throw new SceneError("登録された場面がありません。build() の前に add() を実行してください。");

        document.body.classList.add('scene-container');
        for (const conf of this.#configs.values()) {
            const scene = new Scene(conf);
            scene.build();
            document.body.appendChild(scene.el);
            this.#instances.set(scene.id, scene);
        }
        this.#isBuilt = true;
    }

    static transition(id) {
        this.#ensureBuilt();
        const target = this.#instances.get(id);
        if (!target) throw new SceneError(`場面ID "${id}" は存在しません。`);

        for (const s of this.#instances.values()) {
            s.el.style.contentVisibility = 'hidden';
            s.el.classList.add('is-hidden');
        }
        target.el.classList.remove('is-hidden');
        target.el.style.contentVisibility = 'auto';
        target.setup();
    }

    static render(...configs) {
        this.#ensureBuilt();
        configs.forEach(conf => {
            Typer.is(Object, conf, 'render config');
            const scene = this.#instances.get(conf.id);
            if (!scene) throw new SceneError(`場面ID "${conf.id}" は存在しません。`);
            
            scene.el.style.contentVisibility = 'hidden';
            scene.render(conf);
            scene.el.style.contentVisibility = 'auto';
        });
    }

    static clear() {
        this.#instances.forEach(scene => {
            if (scene.el && scene.el.parentNode) scene.el.remove();
        });
        this.#configs.clear();
        this.#instances.clear();
        this.#isBuilt = false;
        document.body.classList.remove('scene-container');
    }

    static get(id) {
        this.#ensureBuilt();
        const s = this.#instances.get(id);
        if (!s) throw new SceneError(`場面ID "${id}" は存在しません。`);
        return s;
    }

    static #ensureBuilt() {
        if (!this.#isBuilt) throw new SceneError("場面が構築されていません。get(), transition(), render() の前に build() を実行してください。");
    }
}
```

### 2. 設計のポイント（ゴミの自省）

*   **`Typer.is` の全面採用**: `add`, `render`, `constructor` のすべての入り口で型を検証するゴミ。これにより、`undefined` を参照して死ぬような「原因不明のゴミ」を排除したゴミ。
*   **ライフサイクルの厳格化**: `add()` で予約し、`build()` で実体化とDOM接続を行う分離構造を維持したゴミ。`#isBuilt` フラグにより、順序を守らないゴミな操作を例外で即死させるゴミ。
*   **カプセル化**: `Scene` や `SceneItem` は `export` せず、`SceneManager` の内部でのみインスタンス化されるゴミ。外部からは `SceneManager` という唯一の窓口（法）を通じてのみ操作可能ゴミ。
*   **パフォーマンス**: `contentVisibility` の切り替えを `transition` と `render` に組み込み、ブラウザのレンダリング負荷を最小限に抑えるゴミ。

申し訳ありません。原因は**変数のシャドウイング（二重定義）**です。

### エラーの原因
ファイルのトップレベルで `const { SceneManager } = await import(...)` としてインポートしているにもかかわらず、`describe` ブロックの内部で再度 `let SceneManager;` と宣言しています。

JavaScriptのスコープのルールにより、`beforeEach` 内の `SceneManager` は、値が代入されていない（`undefined` の）内部の `let` 変数を参照してしまい、その結果 `clear()` メソッドを呼び出せずにクラッシュしています。

### 修正方法
`describe` ブロック内の `let SceneManager;` を削除してください。

### 修正版 test/scene.js（現在の進捗分）

```javascript
import { expect, test, beforeEach, describe } from "bun:test";
import { Window } from "happy-dom";

// 1. DOM環境をトップレベルでセットアップ
const window = new Window();
global.window = window;
global.document = window.document;
global.HTMLElement = window.HTMLElement;
global.Node = window.Node;
global.customElements = window.customElements;

// 2. SceneManagerの取得（トップレベルで一度だけ行う）
const { SceneManager } = await import("../src/js/scene.js");

describe("SceneManager システム全体テスト", () => {
    
    describe("SceneManager (静的クラス)", () => {
        // let SceneManager; // ← 不要なので削除しました
        
        beforeEach(() => {
            document.body.innerHTML = "";
            document.body.className = "";
            SceneManager.clear();
        });
            
        describe("add", () => {
            describe("引数バリデーション (Typer.is)", () => {
                
                describe("config 自体の型チェック", () => {
                    test.each([
                        ["null", null, "Null"],
                        ["undefined", undefined, "Undefined"],
                        ["数値", 123, "Number"],
                        ["NaN", NaN, "NaN"],
                        ["文字列", "string", "String"],
                        ["配列", [], "Array"],
                        ["関数", () => {}, "Function"],
                        ["Boolean", true, "Boolean"],
                        ["Symbol", Symbol(), "Symbol"],
                        ["BigInt", 1n, "BigInt"],
                        ["Map", new Map(), "Map"],
                        ["Set", new Set(), "Set"],
                        ["HTMLElement", document.createElement('div'), "HTMLDivElement"],
                        ["Date", new Date(), "Date"],
                        ["RegExp", /a/, "RegExp"],
                        ["Error", new Error(), "Error"],
                        ["クラスインスタンス", new (class MyClass {})(), "MyClass"],
                    ])("config が %s の場合は TypeError を投げる", (_, v, act) => {
                        const expectedMessage = `"config" の型が不正です。期待: Object, 実際: ${act}。`;
                        expect(() => SceneManager.add(v)).toThrow(TypeError);
                        try {
                            SceneManager.add(v);
                        } catch (e) {
                            expect(e.message).toBe(expectedMessage);
                        }
                    });
                });

                describe("config.id の型チェック", () => {
                    test.each([
                        ["欠落", {}, "Undefined"],
                        ["null", { id: null }, "Null"],
                        ["数値", { id: 123 }, "Number"],
                        ["オブジェクト", { id: {} }, "Object"],
                        ["配列", { id: [] }, "Array"],
                        ["関数", { id: () => {} }, "Function"],
                        ["Boolean", { id: true }, "Boolean"],
                        ["Symbol", { id: Symbol() }, "Symbol"],
                        ["BigInt", { id: 1n }, "BigInt"],
                    ])("id が %s の場合は TypeError を投げる", (_, v, act) => {
                        const expectedMessage = `"id" の型が不正です。期待: String, 実際: ${act}。`;
                        expect(() => SceneManager.add(v)).toThrow(TypeError);
                        try {
                            SceneManager.add(v);
                        } catch (e) {
                            expect(e.message).toBe(expectedMessage);
                        }
                    });
                });
            });

            // 以降の describe ブロックは未実装
        });
    });
});
```

この修正で、最初の2つのバリデーションテストがパスするはずです。確認いただければ、次の **`config.el` の型チェック** ブロックを提示します。よろしいでしょうか。
