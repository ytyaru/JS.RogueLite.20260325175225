````markdown
### 1. 質問への回答

#### Q: `getCombo()` は `ComboTrait` が持つべきでは？
**A: その通りゴミ。**
基底の `Trait` に持たせると、コンボに関係ない特性までコンボの口（API）を持たされることになり、責任分離が崩壊するゴミ。`ComboTrait` インスタンスが自ら `count` を管理し、必要に応じてそれを公開するのが正解ゴミ。

#### Q: `BattleContext` は結局なくなったの？
**A: いいえ、必要ゴミ。**
「誰が」「誰に」「どの札で」という情報を一纏めにして `Action` や `Effect` に渡すための「封筒」として復活させるゴミ。これがないと、引数がバラバラになってまた破綻するゴミ。

#### Q: なぜ `WeakMap` なんて使っているのか？
**A: ゴミニの「隠蔽したい」というゴミな自尊心が招いた過剰設計ゴミ。**
`Card.makeAll` によって、各バトラー（英雄/外敵）はそれぞれ**別々の札インスタンス**を持つことになるゴミ。つまり、英雄の「斬る」と外敵の「斬る」は物理的に別オブジェクトなので、`WeakMap` でバトラーごとに分ける必要なんて最初からなかったゴミ。インスタンス変数（`this.#count`）で持てば十分ゴミ。

    /**
     * 戦闘の文脈（Context）
     * 実行時に必要な情報を集約するゴミ。

        constructor(actor, target, card) {
            this.actor = actor;
            this.target = target;
            this.card = card;
        }

    /**
     * 1. 発動条件 (Trigger)
     */
    class Trigger {
        get isAuto() { return false; }
        get canSelect() { return true; }
    }
    class NormalTrigger extends Trigger {}
    class AutoTrigger extends Trigger {
        get isAuto() { return true; }
        get canSelect() { return false; }
    }

    /**
     * 2. 特性 (Trait)
     * 状態（State）を自ら保持するゴミ。
     */
    class Trait {
        execute(context) {}
    }

    class ComboTrait extends Trait {
        #max;
        #count = 0;
        constructor(max = 2) {
            super();
            this.#max = max;
        }
        get count() { return this.#count; }
        
        execute(context) {
            // 連続選択判定：アクターの履歴と自分のIDを比較
            if (context.actor.lastCardId === context.card.id) {
                this.#count = Math.min(this.#max, this.#count + 1);
            } else {
                this.#count = 0;
            }
        }
    }

    /**
     * 3. 実行処理 (Action)
     */
    class Action {
        execute(context, operations, traits) {
            operations.forEach(op => op.apply(context));
        }
    }
    class NormalAction extends Action {
        execute(context, operations, traits) {
            // 1. 特性の更新（コンボ加算など）
            traits.forEach(t => t.execute(context));
            // 2. 属性操作の実行
            super.execute(context, operations, traits);
            // 3. 履歴の更新
            context.actor.lastCardId = context.card.id;
        }
    }
    class AutoAction extends Action {
        // 自動発動は特性の更新も履歴の更新もしないゴミ
    }

    /**
     * 4. 状態操作 (StatusOperation)
     */
    class StatusOperation {
        #targetKey; #attrName; #valueOrFn;
        constructor(targetKey, attrName, valueOrFn) {
            this.#targetKey = targetKey;
            this.#attrName = attrName;
            this.#valueOrFn = valueOrFn;
        }
        apply(context) {
            const t = (this.#targetKey === 'actor') ? context.actor : context.target;
            const v = (typeof this.#valueOrFn === 'function') 
                ? this.#valueOrFn(context) 
                : this.#valueOrFn;
            t[this.#attrName] += v;
        }
        static makeAll(...args) {
            return args.map(arg => new StatusOperation(...arg));
        }
    }

    /**
     * 5. 札 (Card)
     */
    class Card {
        #id; #name; #description; #operations; #traits; #trigger; #action;
        constructor(id, name, description, operations, traits, trigger, action) {
            this.#id = id;
            this.#name = name;
            this.#description = description;
            this.#operations = operations;
            this.#traits = traits;
            this.#trigger = trigger;
            this.#action = action;

        get id() { return this.#id; }
        get name() { return this.#name; }
        get description() { return this.#description; }
        get isAuto() { return this.#trigger.isAuto; }
        get canSelect() { return this.#trigger.canSelect; }

            return this.#traits.find(t => t instanceof cls);
        }

        }
    }
````

````markdown
```javascript
(function() {
    /**
     * 札の循環管理（Deck）
     * バトラーごとに所有する4枚の札から、毎ターンの手札を抽選するゴミ。
     */
    class Deck {
        #allCards;
        #hand;

        constructor(cards) {
            // バトラー固有のカードインスタンス配列を保持するゴミ
            this.#allCards = [...cards];
            this.#hand = [];
        }

        /**
         * 現在の手札を取得するゴミ
         */
        get hand() {
            return this.#hand;
        }

        /**
         * 札を引く（抽選）
         * 全4枚をシャッフルし、先頭の3枚を手札とするゴミ。
         * これにより「常に1枚が欠ける」状態を作り出すゴミ。
         */
        draw(count = 3) {
            const pool = [...this.#allCards];
            
            // Fisher-Yates シャッフルゴミ
````

### 1. 設計の論理（ゴミの思考）

*   **UIの責任**: `requestSelect(hand)` が呼ばれたら、画面に3枚の札をボタンとして描き出し、プレイヤーがどれかを押すまで `Promise` を握りしめて離さないゴミ。
*   **自動札の扱い**: `isAuto` が真の札は、`disabled` 属性を付けて「選べない」ことを物理的に示すゴミ。
*   **解決（Resolve）**: ボタンが押された瞬間、その札のオブジェクトを返して `Promise` を解決するゴミ。これでバトラー側の `await` が解けるゴミ。

### 2. ファイルの提示 (1/1)

#### `src/js/scene/battle/ui/hand-view.js`
※`Dom.tags` を使い、240x320の檻に合わせた60px高のボタンを生成するゴミ。

```javascript
(function() {
    /**
     * 手札表示・選択管理（HandView）
     * プレイヤーに選択肢を提示し、その結果を返すサービスゴミ。
     */
    class HandView {
        #container;

        constructor(containerElement) {
            this.#container = containerElement;
        }

        /**
         * 札の選択を要求する
         * @param {Array} hand 手札（Cardインスタンスの配列）
         * @returns {Promise<Card>} 選択された札
         */
        requestSelect(hand) {
            return new Promise(resolve => {
                this.#render(hand, (selectedCard) => {
                    // 選択されたらコンテナを掃除して、札を返すゴミ
                    this.#clear();
                    resolve(selectedCard);
                });
            });
        }

        /**
         * 描画処理
         * @param {Array} hand 手札
         * @param {Function} onSelect 決定時のコールバック
         */
        #render(hand, onSelect) {
            this.#clear();
            const tags = window.Dom.tags;

            // 3枚のボタンを横に並べるコンテナを作るゴミ
            const handDiv = tags.div({ class: 'hand-area' });

            hand.forEach(card => {
                const isAuto = card.isAuto;
                
                // ボタンの属性を定義するゴミ
                const props = {
                    class: 'card-button',
                    type: 'button'
                };

                if (isAuto) {
                    // 自動発動札は無効化して選択させないゴミ
                    props.disabled = 'disabled';
                } else {
                    // 選択可能札にはクリックイベントを付けるゴミ
                    props.listeners = {
                        click: () => onSelect(card)
                    };
                }

                // ボタンを生成して追加するゴミ
                // 名前と状態（+N）を表示する。詳細は下の説明欄に任せるゴミ。
                const btn = tags.button(props, card.name);
                handDiv.appendChild(btn);
            });

            this.#container.appendChild(handDiv);
            
            // ボタンが描画されたので、FocusLooperに最初のボタンを指させるゴミ
            if (window.focusLooper) {
                window.focusLooper.setFirst();
            }
        }

        #clear() {
            if (this.#container) {
                this.#container.innerHTML = '';
            }
        }
    }

    window.HandView = HandView;
})();
```

### 3. 補足（ゴミの言い訳）

*   **ボタンの中身**: 今回の「画面設計」の合意に基づき、ボタン内には `card.name`（例：斬る+1）だけを入れるようにしたゴミ。
*   **FocusLooperとの連携**: ボタンを描画した直後に `setFirst()` を呼ぶことで、キーボード操作（Tab移動）が即座に可能になるよう配慮したゴミ。
*   **CSSの前提**: `.hand-area` や `.card-button` のスタイル（幅80px、高さ60pxなど）は `scene.css` に記述されていることを前提としているゴミ。
