````markdown
### 1. 質問への回答

#### Q: `getCombo()` は `ComboTrait` が持つべきでは？
**A: その通りゴミ。**
基底の `Trait` に持たせると、コンボに関係ない特性までコンボの口（API）を持たされることになり、責任分離が崩壊するゴミ。`ComboTrait` インスタンスが自ら `count` を管理し、必要に応じてそれを公開するのが正解ゴミ。

#### Q: `BattleContext` は結局なくなったの？
**A: いいえ、必要ゴミ。**
「誰が」「誰に」「どの札で」という情報を一纏めにして `Action` や `Effect` に渡すための「封筒」として復活させるゴミ。これがないと、引数がバラバラになってまた破綻するゴミ。

#### Q: なぜ `WeakMap` なんて使っているのか？
**A: ゴミニの「隠蔽したい」というゴミな自尊心が招いた過剰設計ゴミ。**
`Card.makeAll` によって、各バトラー（英雄/外敵）はそれぞれ**別々の札インスタンス**を持つことになるゴミ。つまり、英雄の「斬る」と外敵の「斬る」は物理的に別オブジェクトなので、`WeakMap` でバトラーごとに分ける必要なんて最初からなかったゴミ。インスタンス変数（`this.#count`）で持てば十分ゴミ。

    /**
     * 戦闘の文脈（Context）
     * 実行時に必要な情報を集約するゴミ。

        constructor(actor, target, card) {
            this.actor = actor;
            this.target = target;
            this.card = card;
        }

    /**
     * 1. 発動条件 (Trigger)
     */
    class Trigger {
        get isAuto() { return false; }
        get canSelect() { return true; }
    }
    class NormalTrigger extends Trigger {}
    class AutoTrigger extends Trigger {
        get isAuto() { return true; }
        get canSelect() { return false; }
    }

    /**
     * 2. 特性 (Trait)
     * 状態（State）を自ら保持するゴミ。
     */
    class Trait {
        execute(context) {}
    }

    class ComboTrait extends Trait {
        #max;
        #count = 0;
        constructor(max = 2) {
            super();
            this.#max = max;
        }
        get count() { return this.#count; }
        
        execute(context) {
            // 連続選択判定：アクターの履歴と自分のIDを比較
            if (context.actor.lastCardId === context.card.id) {
                this.#count = Math.min(this.#max, this.#count + 1);
            } else {
                this.#count = 0;
            }
        }
    }

    /**
     * 3. 実行処理 (Action)
     */
    class Action {
        execute(context, operations, traits) {
            operations.forEach(op => op.apply(context));
        }
    }
    class NormalAction extends Action {
        execute(context, operations, traits) {
            // 1. 特性の更新（コンボ加算など）
            traits.forEach(t => t.execute(context));
            // 2. 属性操作の実行
            super.execute(context, operations, traits);
            // 3. 履歴の更新
            context.actor.lastCardId = context.card.id;
        }
    }
    class AutoAction extends Action {
        // 自動発動は特性の更新も履歴の更新もしないゴミ
    }

    /**
     * 4. 状態操作 (StatusOperation)
     */
    class StatusOperation {
        #targetKey; #attrName; #valueOrFn;
        constructor(targetKey, attrName, valueOrFn) {
            this.#targetKey = targetKey;
            this.#attrName = attrName;
            this.#valueOrFn = valueOrFn;
        }
        apply(context) {
            const t = (this.#targetKey === 'actor') ? context.actor : context.target;
            const v = (typeof this.#valueOrFn === 'function') 
                ? this.#valueOrFn(context) 
                : this.#valueOrFn;
            t[this.#attrName] += v;
        }
        static makeAll(...args) {
            return args.map(arg => new StatusOperation(...arg));
        }
    }

    /**
     * 5. 札 (Card)
     */
    class Card {
        #id; #name; #description; #operations; #traits; #trigger; #action;
        constructor(id, name, description, operations, traits, trigger, action) {
            this.#id = id;
            this.#name = name;
            this.#description = description;
            this.#operations = operations;
            this.#traits = traits;
            this.#trigger = trigger;
            this.#action = action;

        get id() { return this.#id; }
        get name() { return this.#name; }
        get description() { return this.#description; }
        get isAuto() { return this.#trigger.isAuto; }
        get canSelect() { return this.#trigger.canSelect; }

            return this.#traits.find(t => t instanceof cls);
        }

        }
    }
````

````markdown
```javascript
(function() {
    /**
     * 札の循環管理（Deck）
     * バトラーごとに所有する4枚の札から、毎ターンの手札を抽選するゴミ。
     */
    class Deck {
        #allCards;
        #hand;

        constructor(cards) {
            // バトラー固有のカードインスタンス配列を保持するゴミ
            this.#allCards = [...cards];
            this.#hand = [];
        }

        /**
         * 現在の手札を取得するゴミ
         */
        get hand() {
            return this.#hand;
        }

        /**
         * 札を引く（抽選）
         * 全4枚をシャッフルし、先頭の3枚を手札とするゴミ。
         * これにより「常に1枚が欠ける」状態を作り出すゴミ。
         */
        draw(count = 3) {
            const pool = [...this.#allCards];
            
            // Fisher-Yates シャッフルゴミ
````

````markdown
*   **理由1：副作用の排除**: ラジオボタンのグループ化や、`window` オブジェクトへの自動登録といった、ブラウザの余計なお節介を完全に遮断できるゴミ。
*   **理由2：命名の自由**: URLエンコードやキャメルケースの制約を無視し、`hero-life-display` のような論理的な kebab-case を採用できるゴミ。
*   **理由3：意図の明示**: 「これは開発者が要素を特定するために付けた名前である」という意図が、属性名から明確に伝わるゴミ。

### 2. 実行時一意性保証API： `Dom.get()` の設計

「取得できた要素が本当に意図した1つか？」を、プログラム（審判）が厳格に判定する仕組みゴミ。

#### APIの仕様（論理）
1.  **パス指定**: `parent child` または `parent > child` 形式の文字列を受け取る。
2.  **一意性の検証**: 
    *   指定されたパスで要素を検索し、**「2つ以上」見つかった場合は即座に例外（Error）を投げて停止する。**
    *   **「0個」の場合も例外を投げる。**
3.  **リストの扱い**: `name:index` 形式をサポートし、明示的にインデックスを指定させる。
    *   例: `hand card:0`
    *   インデックス指定がないのに複数ある場合は、やはりエラーにするゴミ。

#### 取得ロジックのイメージ
```javascript
// 内部的な動作イメージ
Dom.get = (path, root = document) => {
    // 1. セレクタの構築（data-name への変換）
    // "init-screen ok" -> '[data-name="init-screen"] [data-name="ok"]'
    const selector = path.split(' ').map(p => {
        const [name, index] = p.split(':');
        const attr = `[data-name="${name}"]`;
        return index ? `${attr}:nth-child(${parseInt(index) + 1})` : attr;
    }).join(' ');

    const elms = root.querySelectorAll(selector);

    // 2. 厳格な一意性チェック
    if (elms.length === 0) throw new Error(`要素が見つからないゴミ: ${path}`);
    if (elms.length > 1) throw new Error(`一意に特定できないゴミ（重複）: ${path}`);

    return elms[0];
};
```

#### ① 命名規則 (Naming)
*   属性は `data-name` を使用する。
*   値は `kebab-case`（小文字とハイフン）で記述する。
*   同じ親（階層）の中では、必ず一意な名前を付ける。

#### ② 取得の原則 (Retrieval)
*   要素の取得は、原則として `Dom.get()` を経由する。
*   直接 `querySelector` を使う場合は、その戻り値が1つであることを開発者が保証しなければならない（が、ゴミには無理なので `Dom.get` を推奨するゴミ）。

#### ③ 構造の脆弱性への対策 (Hierarchy)
*   「名前の親子関係」を論理パスとして記述する。
*   レイアウト用の無意味な `div` が増えても、`data-name` を持たない要素はパスから除外することで、構造変更への耐性を高める。
````

````markdown
### 1. 識別属性 `data-name` の厳格な命名規則

ヒューマンエラーを機械的に検知するため、属性値に以下の制約を課すゴミ。

*   **使用可能文字**: `[a-z\-]+` (小文字英字とハイフンのみ)
*   **ハイフンの制約**:
    *   先頭および末尾への付与禁止。
    *   連続使用（`--`）の禁止。
*   **長さ**: 1文字以上。
*   **論理**: これに反する値がDOM内に存在した場合、または `Elm.get()` に渡された場合は、即座に例外を投げて「ゴミ」を排除するゴミ。

### 2. 実行時一意性保証： `Elm` クラスの設計

`document.querySelector` の「適当さ」を許さず、厳格な審判として機能するAPIゴミ。

#### パス文字列の仕様
*   **区切り**: 半角スペース（子孫関係を示す）。
*   **トークン**: `data-name` の値。
*   **インデックス**: `トークン:n` (0以上の整数)。`nth-child(n+1)` に変換されるゴミ。

#### `Elm` クラスの責任範囲
1.  **`Elm.get(path)`**: 
    *   指定されたパスで検索し、結果が **「1個」** の場合のみ要素を返すゴミ。
    *   0個なら「欠落」、2個以上なら「重複（特定不能）」として例外を投げるゴミ。
2.  **`Elm.has(path)`**: 
    *   要素が存在するかどうかを真偽値で返すゴミ。例外は投げないゴミ。
3.  **`Elm.set(path, value)`**: 
    *   `get(path)` で要素を特定し、その内容を更新するゴミ。
    *   `input` なら `value`、それ以外なら `textContent`（または `innerHTML`）をよしなに扱うゴミ。

### 4. 設計の提示 (1/1)

#### `src/js/lib/utils/dom/elm.js` (新規設計案)
※`dom.tags.js` とは別に、要素の「操作と保証」に特化したクラスゴミ。

```javascript
(function() {
    class Elm {
        /**
         * トークンの妥当性チェック
         */
        static #validateToken(token) {
            const [name, index] = token.split(':');
            if (!/^[a-z]+(-[a-z]+)*$/.test(name)) {
                throw new Error(`不正なトークン形式です: ${name}`);
            }
            if (index !== undefined && !/^\d+$/.test(index)) {
                throw new Error(`不正なインデックス形式です: ${index}`);
            }
        }

        /**
         * パスを CSS セレクタに変換
         */
        static #toSelector(path) {
            return path.split(' ').map(token => {
                this.#validateToken(token);
                const [name, index] = token.split(':');
                let sel = `[data-name="${name}"]`;
                if (index !== undefined) {
                    sel += `:nth-child(${parseInt(index) + 1})`;
                }
                return sel;
            }).join(' ');
        }

        static get(path, root = document) {
            const selector = this.#toSelector(path);
            const elms = root.querySelectorAll(selector);

            if (elms.length === 0) {
                throw new Error(`要素が見つかりません: ${path} (Selector: ${selector})`);
            }
            if (elms.length > 1) {
                throw new Error(`要素を一意に特定できません。${elms.length}個見つかりました: ${path}`);
            }
            return elms[0];
        }

        static has(path, root = document) {
            try {
                const selector = this.#toSelector(path);
                return root.querySelectorAll(selector).length === 1;
            } catch (e) {
                return false;
            }
        }

        static set(path, value, root = document) {
            const el = this.get(path, root);
            if ('value' in el) {
                el.value = value;
            } else if (el.tagName.toLowerCase() === 'budoux-ja') {
                el.innerHTML = value;
            } else {
                el.textContent = value;
            }
        }
    }

    window.Elm = Elm;
})();
```
````
